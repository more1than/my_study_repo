C语言中\n的值是10

scanf函数只会拿去空格前的值

```c
int main()
{
    // system("chcp 65001");
    int ret = 0;
    char password[20] = {0};
    printf("请输入密码:");
    scanf("%s", password); // 输入密码并存入数组
    // 缓冲区还剩余一个回车字符 \n
    printf("请确认(Y/N)");
    int ch = 0;
    while ((ch = getchar())!='\n') // 循环清除缓冲区 ，拿走回车符
         {
            ;
         }
    ret = getchar(); // getchar会获取到用户输入的回车\n导致直接进入判断中，因此要清空scanf函数的缓存区
    if (ret == 'Y')
        printf("确认成功\n");
    else
        printf("放弃确认\n");
    return 0;
}
```
