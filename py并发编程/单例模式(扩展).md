### 单例模式

每次实例化类的对象时，都是最开始创建的那个对象，不再重复创建对象

```python
def Dan:
    instance = None
    def __init__(self, name):
        self.name = name
    def __new__(cls, *args, **kwargs):
        if cls.instance:
            return cls.instance
        cls.instance = object.__new__(cls, **args, **kwargs):
        return cls.instance

obj = Dan("zhang")
obj = Dan("hao")
```

多线程的单例模式，可能会出现单例失效的问题，即每次创建的类对象内存地址出现不一样的情况，或因为运行稍慢导致

```python
import threading
import time


class Dan:
    istance = None

    def __init__(self, name):
        self.name = name

    def __new__(cls, *args, **kwargs):
        if cls.istance:
            return cls.istance
        time.sleep(0.1)
        cls.istance = object.__new__(cls)
        print(cls.istance)
        return cls.istance

def task():
    obj = Dan("zhang")
    print(obj)

for i in range(10):
    t = threading.Thread(target=task)
    t.start()
    
# 运行结果
<__main__.Dan object at 0x0000017023F6A160>
<__main__.Dan object at 0x0000017023F6A160>
<__main__.Dan object at 0x000001702413F9B0>
<__main__.Dan object at 0x0000017024131438>
<__main__.Dan object at 0x0000017024131D30>
```

优化方案

```python
class Dan:
    istance = None
    lock = threading.RLock()

    def __init__(self, name):
        self.name = name

    def __new__(cls, *args, **kwargs):
        # 如果类对象存在，跳过申请和释放锁的操作，节省资源
        if cls.istance:
            return cls.istance
        # 添加线程锁，只有一个线程能申请到锁
        with cls.lock:
            if cls.istance:
                return cls.istance
            cls.istance = object.__new__(cls)
            print(cls.istance)
            return cls.istance

def task():
    obj = Dan("zhang")
    print(obj)

for i in range(10):
    t = threading.Thread(target=task)
    t.start()
```
